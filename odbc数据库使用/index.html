#include <iostream>
//ODBC用到的头文件
#include <windows.h>
#include <sql.h>
#include <sqlext.h>
#include <sqltypes.h>
#include <tchar.h>
#include <io.h>      // 包含 _setmode
#include <fcntl.h>   // 包含 _O_U16TEXT
using namespace std;

void check(SQLRETURN ret, wstring type, SQLHENV hEnv, SQLHDBC hDbc, SQLHSTMT hStmt) {
    if (ret == SQL_ERROR) {
        wcout << type << _T("失败！") << endl;
        //获取错误信息 
        SQLTCHAR state[128] = { 0 };
        SQLTCHAR msg[128] = { 0 };
        SQLRETURN ret = SQLError(hEnv, hDbc, hStmt, state, NULL, msg, sizeof(msg), NULL);
        wcout << state << " " << msg << endl;

        exit(-1);
    }
}

int main()
{
    //设置为中文兼容unicode
    _wsetlocale(LC_ALL, L"chs");

    // 设置控制台输出编码为UTF-8
    _setmode(_fileno(stdout), _O_U16TEXT);

    // 声明环境句柄
    SQLHENV hEnv = NULL;
    //声明连接句柄
    SQLHDBC hDbc = NULL;
    //声明语句句柄
    SQLHSTMT hStmt = NULL;
    //声明返回值
    SQLRETURN ret;

    // SQL_HANDLE_ENV;  sql handle environment 环境
    // SQL_HANDLE_DBC; sql handle database connection 数据库连接
    // SQL_HANDLE_STMT; sql handle statement 执行sql语句
    // SQL_HANDLE_DESC; sql handle descriptor 描述数据的属性
    
    //分配环境句柄
    ret = SQLAllocHandle(SQL_HANDLE_ENV, SQL_NULL_HANDLE, &hEnv); 
    check(ret, _T("分配环境句柄"), hEnv, hDbc, hStmt);

    //设定ODBC版本 sql set environment attribute
    ret = SQLSetEnvAttr(hEnv, SQL_ATTR_ODBC_VERSION, (SQLPOINTER)SQL_OV_ODBC3, 0);
    check(ret, _T("ODBC版本设置"), hEnv, hDbc, hStmt);
    
    //分配数据库连接句柄
    ret = SQLAllocHandle(SQL_HANDLE_DBC, hEnv, &hDbc);
    check(ret, _T("分配连接句柄"), hEnv, hDbc, hStmt);
    
    //根据DSN连接数据库，SQL_NTS为自动计算前面字符串长度
    // 使用SQLDriverConnect并指定连接字符串
    SQLTCHAR connectionString[] = _T("Driver={SQL Server};Server=172.16.10.251,10097;Database=gameconfig;Uid=fz536;Pwd=G4H7cA3pILjo;");
    //sql type char
    SQLTCHAR outConnString[1024];
    SQLSMALLINT outConnStringLen;

    //SQL_NTS是sql null terminated string
    ret = SQLDriverConnect(hDbc, NULL, connectionString,
        SQL_NTS, outConnString, 1024, &outConnStringLen, SQL_DRIVER_NOPROMPT);

    check(ret, _T("连接数据库"), hEnv, hDbc, hStmt);
    wcout << _T("连接数据库成功！") << endl;

    // 分配语句句柄
    ret = SQLAllocHandle(SQL_HANDLE_STMT, hDbc, &hStmt);
    check(ret, _T("分配语句句柄"), hEnv, hDbc, hStmt);

    // sql查询语句
    SQLTCHAR sql[] = _T("SELECT 道具名称 FROM gameconfig.dbo.gcf_item WHERE 道具名称 LIKE '%剑%'");
    ret = SQLExecDirect(hStmt, sql, SQL_NTS); // SQL_NTS为自动计算前面字符串长度
    check(ret, _T("执行查询语句"), hEnv, hDbc, hStmt);

    // 绑定列
    TCHAR name[128] = { 0 };
    SQLLEN len;

    // SQLRETURN SQLBindCol(
    // SQLHSTMT       StatementHandle,  // 语句句柄
    // SQLUSMALLINT   ColumnNumber,     // 列号，从1开始
    // SQLSMALLINT    TargetType,       // 目标类型，表示应用程序中存储数据的类型
    // SQLPOINTER     TargetValuePtr,   // 目标指针，指向存储数据的缓冲区
    // SQLLEN         BufferLength,     // 缓冲区长度
    // SQLLEN        *StrLen_or_Ind     // 指向实际数据长度或指示器的指针
    // );
    // 这段代码绑定了结果集的第一列到 name 缓冲区。
    // 当我们调用 SQLFetch 时，第一列的数据会被自动放入 name 缓冲区。
    
    ret = SQLBindCol(hStmt, 1, SQL_C_WCHAR, name, sizeof(name), &len);
    check(ret, _T("绑定列"), hEnv, hDbc, hStmt);

    ret = SQLFetch(hStmt);
    while (ret != SQL_NO_DATA)
    {
        wcout << name << endl;
        ZeroMemory(name, sizeof(name));
        ret = SQLFetch(hStmt);
    }
    
    //释放语句句柄
    if (hStmt) {
        SQLFreeHandle(SQL_HANDLE_STMT, hStmt);
    }
    //释放连接句柄
    if (hDbc) {
        SQLDisconnect(hDbc);
        SQLFreeHandle(SQL_HANDLE_DBC, hDbc);
    }
    //释放环境句柄
    if (hEnv) {
        SQLFreeHandle(SQL_HANDLE_ENV, hEnv);
    }
    wcout << _T("断开数据库连接！") << endl;
    return 0;
}
