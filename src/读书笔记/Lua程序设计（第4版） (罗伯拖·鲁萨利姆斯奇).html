# 读取写入文件

-- Function to sort lines alphabetically
local function sort_lines(lines)
    table.sort(lines)
    return lines
end

-- Function to read lines from a file or standard input
local function read_lines(filename)
    local lines = {}
    local file
    if filename then
        file = io.open(filename, "r")
    else
        file = io.stdin
    end
    
    for line in file:lines() do
        table.insert(lines, line)
    end
    
    if filename then
        file:close()
    end
    
    return lines
end

-- Function to write lines to a file or standard output
local function write_lines(lines, filename)
    local file
    if filename then
        file = io.open(filename, "w")
    else
        file = io.stdout
    end
    for _, line in ipairs(lines) do
        file:write(line, "\n")
    end
      
    if filename then
        file:close()
    end
end

-- Main function
local function main(...)
    local args = {...}
    local input_filename, output_filename
    args = args[1]
    if args[1] then
        output_filename = args[1]
    elseif args[2] then
        output_filename = args[1]
        input_filename = args[2]
    end
    local lines = read_lines(input_filename)
    local sorted_lines = sort_lines(lines)
    write_lines(sorted_lines, output_filename)
end

function derivative(f, delta)
    delta = delta or 0.0001
    return function(x)
        print("x " .. x)
        print("delta" .. delta)
        return (f(x + delta) - f(x)) / delta
    end
end

local count = derivative(math.cos)

# 协程

-- 定义任务调度器
local TaskScheduler = {}
TaskScheduler.__index = TaskScheduler

function TaskScheduler:new()
  local instance = {
    tasks = {},  -- 存储所有任务
    current_task = 1  -- 当前正在执行的任务索引
  }
  setmetatable(instance, self)
  return instance
end

function TaskScheduler:addTask(taskFunc)
  local task = coroutine.create(taskFunc)
  table.insert(self.tasks, task)
end

function TaskScheduler:run()
  while #self.tasks > 0 do
    local task = self.tasks[self.current_task]
    if task then
      local success, message = coroutine.resume(task, self)
      if not success then
        print("Task Error: ", message)
        table.remove(self.tasks, self.current_task)  -- 移除出错的任务
      elseif coroutine.status(task) == "dead" then
        table.remove(self.tasks, self.current_task)  -- 移除已完成的任务
      else
        self.current_task = self.current_task % #self.tasks + 1  -- 切换到下一个任务
      end
    end
  end
end

-- 创建一个示例任务
local function task1(scheduler)
  for i = 1, 3 do
    print("Task 1, step " .. i)
    coroutine.yield()  -- 暂停任务1，交回控制权给调度器
  end
  print("Task 1 completed!")
end

-- 创建另一个示例任务
local function task2(scheduler)
  for i = 1, 5 do
    print("Task 2, step " .. i)
    coroutine.yield()  -- 暂停任务2，交回控制权给调度器
  end
  print("Task 2 completed!")
end

-- 创建第三个示例任务
local function task3(scheduler)
  for i = 1, 2 do
    print("Task 3, step " .. i)
    coroutine.yield()  -- 暂停任务3，交回控制权给调度器
  end
  print("Task 3 completed!")
end

-- 创建任务调度器
local scheduler = TaskScheduler:new()

-- 添加任务到调度器
scheduler:addTask(task1)
scheduler:addTask(task2)
scheduler:addTask(task3)

-- 运行调度器
scheduler:run()

-- 预期输出
-- Task 1, step 1
-- Task 2, step 1
-- Task 3, step 1
-- Task 1, step 2
-- Task 2, step 2
-- Task 3, step 2
-- Task 1, step 3
-- Task 2, step 3
-- Task 1 completed!
-- Task 2, step 4
-- Task 2, step 5
-- Task 2 completed!
-- Task 3 completed!

# 垃圾收集

-- 定义一个模拟的处理大数据的函数
local function processData()
    local data = {}
    for i = 1, 100000 do
      data[i] = "Data " .. i  -- 创建大量的字符串数据
    end
    print("Data processed. Memory used: " .. collectgarbage("count") .. " KB")
    
    -- 返回数据以继续使用
    return data
  end
  
  -- 调用函数并处理数据
  local processedData = processData()
  
  -- 在处理完成后清理数据以释放内存
  processedData = nil  -- 释放对数据的引用
  
  -- 强制进行垃圾收集
  collectgarbage("collect")
  print("Memory after cleanup: " .. collectgarbage("count") .. " KB")
  
  -- 模拟进一步的内存优化
  collectgarbage("setpause", 120)
  collectgarbage("setstepmul", 500)
  
  -- 显示当前的垃圾收集设置
  print("Current garbage collection settings:")
  print("Pause: " .. collectgarbage("setpause"))
  print("Step Multiplier: " .. collectgarbage("setstepmul"))

# Lua中number的取值范围

  • 在Lua5.1及更早的版本之前，number是双精度浮点数

  • 在32位中，number整数的取值范围是-2^31到2^31-1，也就是-2147483648到2147483647

  • 在64位中，number整数的取值范围是-2^63到2^63-1，也就是-9223372036854775808到9223372036854775807

  • 在32位中，number浮点数的取值范围是-3.4E38到3.4E38

  • 在64位中，number浮点数的取值范围是-1.7E308到1.7E308

  • 使用p.var这种形式绑定的时候，小数被自动转换成整数
