import { useState, useEffect } from "react";

function Counter() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    const timer = setInterval(() => {
      console.log("当前计数:", count);
    }, 2000);

    return () => clearInterval(timer); // 清理副作用
  }, [count]);

  return (
    <div>
      <p>计数: {count}</p>
      {/* 当按 setCount 时，count 变化，useEffect 监控 count，执行 useEffect 函数体部分 */}
      <button onClick={() => setCount(count + 1)}>增加</button>
    </div>
  );
}

# 类型拓展

原因：NextAuth 默认的 User 类型只包含 name、email、image 字段，不包含自定义的 role 字段。

解决方案：创建了 types/next-auth.d.ts 类型扩展文件，声明 Session、User 和 JWT 接口包含 role 属性，TypeScript 会自动合并这些类型定义。

import { DefaultSession, DefaultUser } from "next-auth";

declare module "next-auth" {
  interface Session {
    user: {
      role?: string;
    } & DefaultSession["user"];
  }

  interface User extends DefaultUser {
    role?: string;
  }
}

declare module "next-auth/jwt" {
  interface JWT {
    role?: string;
  }
}

CREATE TYPE Human AS (
    name TEXT,
    birthday DATE,
    -- 4乘4的矩阵

    bio JSON,
    -- 键值对存储
    skills Hstore
);

    id SERIAL PRIMARY KEY,
    details Human
);

    id SERIAL PRIMARY KEY,
    details Human
);
