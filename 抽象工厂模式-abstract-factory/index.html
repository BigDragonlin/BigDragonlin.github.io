  3. пҝҪпҝҪпҝҪпҝҪпҝҪЖ·пҝҪпҝҪAbstract ProductпҝҪпҝҪпҝҪпҝҪОӘТ»пҝҪпҝҪпҝҪЖ·пҝҪпҝҪпҝҪпҝҪпҝҪпҝҪпҝҪпҝҪТ»пҝҪпҝҪпҝҪУҝЪЎпҝҪ
  4. пҝҪпҝҪпҝҪпҝҪпҝҪЖ·пҝҪпҝҪConcrete ProductпҝҪпҝҪпҝҪпҝҪКөпҝҪЦіпҝҪпҝҪпҝҪпҝҪЖ·пҝҪпҝҪЙ«пҝҪпҝҪпҝҪпҝҪпҝҪпҝҪДҪУҝЪЈпҝҪпҝҪЙҫпҝҪпҝҪе№ӨпҝҪпҝҪпҝҪпҝҪпҝҪпҝҪпҝҪпҝҪпҝҪпҝҪ

# Step 3: К№пҝҪГіпҝҪпҝҪу№ӨіпҝҪ

пҝҪН»пҝҪпҝҪЛҙпҝҪпҝҪпҝҪпҝҪНЁпҝҪпҝҪпҝҪпҝҪпҝҪу№ӨіпҝҪпҝҪНіпҝҪпҝҪпҝҪпҝҪЖ·пҝҪДҪУҝпҝҪпҝҪл№ӨпҝҪпҝҪпҝҪНІпҝҪЖ·пҝҪпҝҪпҝҪпҝҪпҝҪпҝҪ

void application(GUIFactory& factory) {
    Button* button = factory.createButton();
    Checkbox* checkbox = factory.createCheckbox();
    button->paint();
    checkbox->paint();
    delete button;
    delete checkbox;
}

int main() {
    WindowsFactory windowsFactory;
    LinuxFactory linuxFactory;
    
    std::cout << "Testing WindowsFactory:\n";
    application(windowsFactory);

    std::cout << "\nTesting LinuxFactory:\n";
    application(linuxFactory);

    return 0;
}
