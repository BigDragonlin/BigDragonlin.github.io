# 格式和命名规则

  • 常量用大写和下划线 例如：MAX_SPEED
  • 变量第一个字母小写，小驼峰 例如：playerName
  • 全局变量用g_开头 例如：g_playerName
  • 函数名第一个字母大写，大驼峰 例如：GetPlayerName

Lua没有整数类型，它的Number类型是双精度浮点数，可以表示整数和小数。

# pcall 和 xpcall

pcall和xpcall都是用来捕获异常的，pcall是最简单的捕获异常的方法，xpcall是pcall的升级版，可以自定义错误处理函数。

local function divide(a, b)
    assert(b ~= 0, "Division by zero")
    return a / b
end

local function safeDivide(a, b)
    local status, result = pcall(divide, a, b)
    if status then
        return result
    else
       print("Error: " .. result)
    end
end
print(safeDivide(10, 2)) -- 5
print(safeDivide(10, 0)) -- Error: Division by zero
print("test.lua loaded successfully!")

# Lua中的传值和传引用

  • 传值：数值、字符串和布尔值在传递时会复制其值，函数内部的修改不会影响原始变量。
  • 传引用：表在传递时会传递其引用，函数内部的修改会影响原始变量。
function UpdateTab(Tab)
    if type(Tab) ~= "table" then
        return print("Tab is not a table, type: " .. type(Tab))
    end
    for i = 1, #Tab do
        Tab[i] = Tab[i] + 1
        print("inner" .. i, Tab[i])
    end
end

local Tab = { 1, 2, 3 }
UpdateTab(Tab)
print(Tab[1], Tab[2], Tab[3])

# math.random

math.randomseed用于设置随机数种子，一般我们使用os.time()作为种子，如果为了debug，可以设置一个固定的种子

# string.format

%s和%d分别用于字符串和数字的格式化输出，但是他们没什么具体的区别，都可以用于字符串和数字的输出。 %d可以用于数字整数的输出，%.2f可以用于保留两位小数的输出。

local string = string.format("this year is %s, and %s", 02020.4, "hello")
print(string)

local string2 = string.format("this year is %d, and %s", 02020.3, "hello")
print(string2)

local string3 = string.format("this year is %d, and %s", 02020.5, "hello")
print(string3)

-- this year is 2020.4, and hello
-- this year is 2020, and hello
-- this year is 2020, and hello

# 使用Lua处理表单型数据

将表单保存为csv文件（数值用逗号隔开）

function GetValues(lines)
    local values = {}
    local number = 0
    if lines then
        while string.find(lines, ",") do
            local b, e = string.find(lines, ",")
            number = number + 1
            if b-1 == 0 then
                values[number] = "null"
            else
                values[number] = string.sub(lines, 1, b - 1)
            end
            lines = string.sub(lines, b + 1, string.len(lines))
        end
        number = number + 1
        values[number] = lines
    end
    return number, values
end

function OpenFile(file)
    for line in io.lines(file) do
        local number, values = GetValues(line)
        for i = 1, number do
            print(values[i])
        end
    end
end
OpenFile("./enum_FunctionType.csv")
